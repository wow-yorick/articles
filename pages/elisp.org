* 生成 A-Z 对应索引
#+BEGIN_SRC elisp
(progn
  (insert "\n")
  (setq a 0)
  (setq c ?A)
  (while (< a 26)
    (insert  (format "%d - %c\n" a c))
    (setq a     (+ a 1))
    (setq c      (+  c 1))
    ))
#+END_SRC
